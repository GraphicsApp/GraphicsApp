348  double objMaxX = object.getRightBorder();
349  double objMinY = object.getTopBorder();
350  double objMaxY = object.getBottomBorder();
351 
352  if (objMinX < minX) {
353  minX = objMinX;
354  }
355 
356  if (objMaxX > maxX) {
357  maxX = objMaxX;
358  }
359 
360  if (objMinY < minY) {
361  minY = objMinY;
362  }
363 
364  if (objMaxY > maxY) {
365  maxY = objMaxY;
366  }
367  }
368  return x >= minX && x <= maxX && y >= minY && y <= maxY;
