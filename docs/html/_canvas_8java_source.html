125  private void drawLabel(Graphics2D g2d, Label label) {
126  AffineTransform originalTransform = g2d.getTransform();
127  AffineTransform rotatedTransform = getRotationTransformForObject(label);
128  rotatedTransform.translate(label.getXPos(), label.getYPos());
129  rotatedTransform.scale(1, 1);
130  g2d.setTransform(rotatedTransform);
131  g2d.setColor(label.getColor().asAWTColor());
132  g2d.setFont(new Font(label.getFont(), Font.PLAIN, label.getFontSize()));
133  /**
134  * Keep in mind: drawString's y parameter defines the text's baseline position not the label's upper left corner.
135  * To keep this call consistent with similar drawing operations (see drawRectangle or drawImage) we are
136  * translating the label's origin by using the current font size as an offset for the y position. This is only
137  * an estimate, since font specific measures (top, ascent, descent or bottom) are not considered.
138  */
139  g2d.drawString(label.getText(), label.getXPos(), label.getYPos() + label.getFontSize());
140  g2d.setTransform(originalTransform);
